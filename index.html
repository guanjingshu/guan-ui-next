    # guan-ui-next 组件库

    # 快速开始

    # 1.安装组件库

    npm i guan-ui-next

    # 2.引用组件库

    // 全部引入
import "guan-ui-next/dist/css/index.css";
import GUI from "guan-ui-next";
Vue.use(GUI);

// 按需引入
import "guan-ui-next/dist/css/demo.css";
import Card from "guan-ui-next";
Vue.use(Card);
